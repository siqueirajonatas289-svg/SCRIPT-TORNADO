local function Cleanup()
    -- Desconecta eventos anteriores salvos globalmente
    for _, conn in pairs(getgenv().VortexManager.Connections) do
        if typeof(conn) == "RBXScriptConnection" then
            conn:Disconnect()
        end
    end
    getgenv().VortexManager.Connections = {}

    -- Cancela threads antigas salvas globalmente
    for _, thread in pairs(getgenv().VortexManager.Tasks) do
        task.cancel(thread)
    end
    getgenv().VortexManager.Tasks = {}

    -- Remove instâncias da interface remanescentes
    local container = (gethui and gethui()) or LocalPlayer:WaitForChild("PlayerGui", 5)
    if container then
        local existing = container:FindFirstChild("VortexTornadoGUI")
        if existing then existing:Destroy() end
    end
end
Cleanup()

local function criarControle(parent, titulo, default, yPos, min, max, passo, callback)
    local label = Instance.new("TextLabel", parent)
    label.Size = UDim2.new(1, 0, 0, 15)
    label.Position = UDim2.new(0, 0, 0, yPos)
    label.Text = titulo .. ": " .. default
    label.TextColor3 = Color3.fromRGB(255, 255, 255)
    label.BackgroundTransparency = 1
    label.Font = Enum.Font.SourceSans
    label.TextSize = 13
    
    local btnMenos = Instance.new("TextButton", parent)
    btnMenos.Size = UDim2.new(0, 30, 0, 18)
    btnMenos.Position = UDim2.new(0.2, 0, 0, yPos + 18)
    btnMenos.Text = "<"
    btnMenos.BackgroundColor3 = Color3.fromRGB(50, 45, 60)
    btnMenos.TextColor3 = Color3.fromRGB(255, 255, 255)
    Instance.new("UICorner", btnMenos).CornerRadius = UDim.new(0, 3)
    
    local btnMais = Instance.new("TextButton", parent)
    btnMais.Size = UDim2.new(0, 30, 0, 18)
    btnMais.Position = UDim2.new(0.65, 0, 0, yPos + 18)
    btnMais.Text = ">"
    btnMais.BackgroundColor3 = Color3.fromRGB(50, 45, 60)
    btnMais.TextColor3 = Color3.fromRGB(255, 255, 255)
    Instance.new("UICorner", btnMais).CornerRadius = UDim.new(0, 3)
    
    local val = default
    btnMenos.MouseButton1Click:Connect(function() val = math.max(min, val - passo) label.Text = titulo..": "..val callback(val) end)
    btnMais.MouseButton1Click:Connect(function() val = math.min(max, val + passo) label.Text = titulo..": "..val callback(val) end)

BtnShoot.MouseButton1Click:Connect(function()
    BtnShoot.BackgroundColor3 = Color3.fromRGB(50, 200, 50)
    local Character = LocalPlayer.Character
    if not Character then BtnShoot.BackgroundColor3 = Color3.fromRGB(200, 50, 50) return end
    local RootPart = Character:FindFirstChild("HumanoidRootPart")
    if not RootPart then BtnShoot.BackgroundColor3 = Color3.fromRGB(200, 50, 50) return end
    local look = Camera.CFrame.LookVector
    if ShootMode == "Mouse" then
        if Mouse.Target then look = (Mouse.Hit.Position - RootPart.Position).Unit
        else look = Mouse.UnitRay.Direction end
    elseif ShootMode == "Personagem" then look = RootPart.CFrame.LookVector end
    for part, _ in pairs(ActiveParts) do
        if part and part.Parent then
            local m = part:FindFirstChild("VortexMover")
            local t = part:FindFirstChild("VortexRotator")
            if m then m:Destroy() end
            if t then t:Destroy() end
            part:SetAttribute("ShotByVortex", true)
            local sM = Instance.new("BodyVelocity", part)
            sM.Name = "ShootMover"
            sM.MaxForce = Vector3.new(1e7, 1e7, 1e7)
            sM.Velocity = look * ShootSpeed
            ActiveParts[part] = nil
            task.delay(2.5, function() if sM then sM:Destroy() end part:SetAttribute("ShotByVortex", nil) end)
        end
    end
    task.wait(0.2)
    BtnShoot.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
end)

-- Efeito visual com gradiente de cor realista
local function createDebrisEffect(part)
    if part:FindFirstChild("TornadoDebris") then return end
    local emitter = Instance.new("ParticleEmitter", part)
    emitter.Name = "TornadoDebris"
    emitter.Texture = "rbxassetid://243089343"
    emitter.Lifetime = NumberRange.new(0.5, 1.2)
    emitter.Rate = 12
    emitter.Speed = NumberRange.new(2, 5)
    emitter.Size = NumberSequence.new({
        NumberSequenceKeypoint.new(0, 0.4),
        NumberSequenceKeypoint.new(1, 1.2)
    })
    -- Gradiente: começa cor de terra/marrom e sobe acinzentado/transparente
    emitter.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0, Color3.fromRGB(110, 85, 65)),
        ColorSequenceKeypoint.new(0.5, Color3.fromRGB(90, 90, 90)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(180, 180, 180))
    })
    emitter.Transparency = NumberSequence.new({
        NumberSequenceKeypoint.new(0, 0.3),
        NumberSequenceKeypoint.new(1, 1)
    })
end

-- Rotação caótica em todas as direções (Ideia 2)
local function applyDebrisTorque(part)
    local rotator = part:FindFirstChild("VortexRotator") or Instance.new("BodyAngularVelocity", part)
    rotator.Name = "VortexRotator"
    rotator.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
    rotator.AngularVelocity = Vector3.new(
        math.random(-8, 8),
        math.random(-15, 15),
        math.random(-8, 8)
    )
end

local function ApplyVortexForce(part, centerPos)
    if part.Anchored then return end
    part.CanCollide = false
    createDebrisEffect(part)
    applyDebrisTorque(part)
    
    local mover = part:FindFirstChild("VortexMover") or Instance.new("BodyVelocity", part)
    mover.Name = "VortexMover" mover.MaxForce = Vector3.new(1e6, 1e6, 1e6)
    local state = ActiveParts[part] or {
        direction="GoingUp", 
        angle=math.atan2(part.Position.Z-centerPos.Z, part.Position.X-centerPos.X),
        yOffset = math.random(-5, 10),
        phase = math.random(1, 1000) / 100,
        randomLayer = math.random(1, 5), 
        verticalLayer = math.random(1, 5) 
    }
    ActiveParts[part] = state
    local timeChaos = math.sin(tick()) * ChaosFactor
    local pulseOffset = (PulseActive and (math.sin(PulseTime) * 10) or 0) + timeChaos
    local massMod = math.clamp(1 / (part:GetMass() / 10), 0.5, 2)
    local speedMod = 1
    local currentOrbit = OrbitDistance + pulseOffset
    local finalAngle = state.angle + state.phase
    if SpiralActive then
        local targetY = centerPos.Y + state.yOffset
        if SpiralLayersActive then
            local verticalHeights = {10, 25, 40, 55, 70}
            targetY = centerPos.Y + verticalHeights[state.verticalLayer]
            state.angle += (RotationSpeed * massMod)
        else
            state.angle += (RotationSpeed * massMod)
            state.yOffset += (state.direction == "GoingUp" and 0.8 or -0.8)
            if state.yOffset >= 50 then state.direction = "GoingDown" 
            elseif state.yOffset <= -5 then state.direction = "GoingUp" end
        end
        finalAngle = state.angle + state.phase
        mover.Velocity = (Vector3.new(centerPos.X+math.cos(finalAngle)*currentOrbit, targetY, centerPos.Z+math.sin(finalAngle)*currentOrbit) - part.Position) * 10
    elseif VortexActive then
        if ConcentricActive then
            local speedMultipliers = {2.0, 1.6, 1.2, 0.8, 0.4}
            local distanceMultipliers = {0.3, 0.6, 0.9, 1.2, 1.5}
            local targetOrbit = OrbitDistance * distanceMultipliers[state.randomLayer]
            currentOrbit = targetOrbit + pulseOffset
            speedMod = speedMultipliers[state.randomLayer] * massMod
        else
            speedMod = massMod
        end
        state.angle += (RotationSpeed * speedMod)
        finalAngle = state.angle + state.phase
        mover.Velocity = (Vector3.new(centerPos.X+math.cos(finalAngle)*currentOrbit, centerPos.Y+(math.sin(finalAngle*1.5)*5)+3, centerPos.Z+math.sin(finalAngle)*currentOrbit) - part.Position) * 12
    end
end

local function ApplyHurricaneForce(part, centerPos)
    if part.Anchored then return end
    part.CanCollide = false
    createDebrisEffect(part)
    applyDebrisTorque(part)
    
    local mover = part:FindFirstChild("VortexMover") or Instance.new("BodyVelocity", part)
    mover.Name = "VortexMover"
    mover.MaxForce = Vector3.new(1e6, 1e6, 1e6)
    
    local state = ActiveParts[part] or {
        direction="GoingUp", 
        angle=math.atan2(part.Position.Z-centerPos.Z, part.Position.X-centerPos.X),
        yOffset = math.random(-5, 10),
        phase = math.random(1, 1000) / 100,
        randomLayer = math.random(1, 5), 
        verticalLayer = math.random(1, 5) 
    }
    ActiveParts[part] = state
    
    local dir = (centerPos - part.Position).Unit
    local tangent = Vector3.new(-dir.Z, 0, dir.X)
    mover.Velocity = (dir * (HurricaneSpeed / 3)) + (tangent * HurricaneSpeed) + Vector3.new(0, 10, 0)
end

-- Efeito visual de clarão de raios/relâmpago no ambiente (Ideia 1)
local function triggerLightningFlash()
    if math.random(1, 40) == 1 then
        local originalColor = Lighting.Ambient
        Lighting.Ambient = Color3.fromRGB(200, 225, 255)
        task.delay(0.06, function()
            Lighting.Ambient = Color3.fromRGB(30, 30, 40)
            task.delay(0.04, function()
                Lighting.Ambient = Color3.fromRGB(180, 210, 255)
                task.delay(0.05, function()
                    Lighting.Ambient = originalColor
                end)
            end)
        end)
    end
end

local heartbeatConn = RunService.Heartbeat:Connect(function(dt)
    if PulseActive then PulseTime += dt * 2 end
    ChaosFactor = math.sin(tick() * 0.5) * 5 
    local Character = LocalPlayer.Character
    if not Character then return end
    local RootPart = Character:FindFirstChild("HumanoidRootPart")
    if not RootPart then return end
    
    if VortexActive or SpiralActive or PulseActive or ConcentricActive or SpiralLayersActive or HurricaneActive then
        local radius = HurricaneActive and HurricaneRadius or VortexRadius
        local hitParts = {}
        
        if HurricaneActive then
            triggerLightningFlash()
        end
        
        for _, p in ipairs(Workspace:GetPartBoundsInRadius(RootPart.Position, radius)) do
            if p:IsA("BasePart") and not p:IsDescendantOf(Character) and not p.Anchored and not p:GetAttribute("ShotByVortex") then
                hitParts[p] = true
                if HurricaneActive then
                    ApplyHurricaneForce(p, RootPart.Position)
                else
                    ApplyVortexForce(p, RootPart.Position)
                end
            end
        end
        for p, _ in pairs(ActiveParts) do
            if not hitParts[p] then 
                local m = p:FindFirstChild("VortexMover") 
                local e = p:FindFirstChild("TornadoDebris")
                local r = p:FindFirstChild("VortexRotator")
                if m then m:Destroy() end 
                if e then e:Destroy() end
                if r then r:Destroy() end
                ActiveParts[p] = nil 
            end
        end
    else
        for p, _ in pairs(ActiveParts) do
            local m = p:FindFirstChild("VortexMover")
            local e = p:FindFirstChild("TornadoDebris")
            local r = p:FindFirstChild("VortexRotator")
            if m then m:Destroy() end
            if e then e:Destroy() end
            if r then r:Destroy() end
            ActiveParts[p] = nil
        end
    end
end)
table.insert(getgenv().VortexManager.Connections, heartbeatConn)
